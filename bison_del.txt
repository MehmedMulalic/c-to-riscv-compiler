%%

primary_expression
    : IDENTIFIER                /* x */
    | CONSTANT                  /* 42 */
    | STRING_LITERAL            /* "hello" */
    | '(' expression ')'        /* (a + b) */
    ;

postfix_expression
    : primary_expression
    | postfix_expression '[' expression ']'      /* arr[5] */
    | postfix_expression '(' ')'                 /* func() */
    | postfix_expression '(' argument_expression_list ')' /* func(a, b) */
    | postfix_expression '.' IDENTIFIER          /* obj.field */
    | postfix_expression PTR_OP IDENTIFIER       /* ptr->field */
    | postfix_expression INC_OP                  /* x++ */
    | postfix_expression DEC_OP                  /* x-- */
    ;

argument_expression_list
    : assignment_expression                     /* a */
    | argument_expression_list ',' assignment_expression /* a, b */
    ;

unary_expression
    : postfix_expression
    | INC_OP unary_expression                  /* ++x */
    | DEC_OP unary_expression                  /* --x */
    | unary_operator cast_expression           /* *p */
    | SIZEOF unary_expression                  /* sizeof x */
    | SIZEOF '(' type_name ')'                 /* sizeof(int) */
    ;

unary_operator
    : '&'     /* &x */
    | '*'     /* *p */
    | '+'     /* +x */
    | '-'     /* -x */
    | '~'     /* ~x */
    | '!'     /* !x */
    ;

cast_expression
    : unary_expression
    | '(' type_name ')' cast_expression        /* (int)x */
    ;

multiplicative_expression
    : cast_expression
    | multiplicative_expression '*' cast_expression /* a * b */
    | multiplicative_expression '/' cast_expression /* a / b */
    | multiplicative_expression '%' cast_expression /* a % b */
    ;

additive_expression
    : multiplicative_expression
    | additive_expression '+' multiplicative_expression /* a + b */
    | additive_expression '-' multiplicative_expression /* a - b */
    ;

shift_expression
    : additive_expression
    | shift_expression LEFT_OP additive_expression  /* a << 2 */
    | shift_expression RIGHT_OP additive_expression /* a >> 2 */
    ;

relational_expression
    : shift_expression
    | relational_expression '<' shift_expression   /* a < b */
    | relational_expression '>' shift_expression   /* a > b */
    | relational_expression LE_OP shift_expression /* a <= b */
    | relational_expression GE_OP shift_expression /* a >= b */
    ;

equality_expression
    : relational_expression
    | equality_expression EQ_OP relational_expression /* a == b */
    | equality_expression NE_OP relational_expression /* a != b */
    ;

and_expression
    : equality_expression
    | and_expression '&' equality_expression       /* a & b */
    ;

exclusive_or_expression
    : and_expression
    | exclusive_or_expression '^' and_expression   /* a ^ b */
    ;

inclusive_or_expression
    : exclusive_or_expression
    | inclusive_or_expression '|' exclusive_or_expression /* a | b */
    ;

logical_and_expression
    : inclusive_or_expression
    | logical_and_expression AND_OP inclusive_or_expression /* a && b */
    ;

logical_or_expression
    : logical_and_expression
    | logical_or_expression OR_OP logical_and_expression /* a || b */
    ;

conditional_expression
    : logical_or_expression
    | logical_or_expression '?' expression ':' conditional_expression /* a ? b : c */
    ;

assignment_expression
    : conditional_expression
    | unary_expression assignment_operator assignment_expression /* a = b */
    ;

assignment_operator
    : '='           /* = */
    | MUL_ASSIGN    /* *= */
    | DIV_ASSIGN    /* /= */
    | MOD_ASSIGN    /* %= */
    | ADD_ASSIGN    /* += */
    | SUB_ASSIGN    /* -= */
    | LEFT_ASSIGN   /* <<= */
    | RIGHT_ASSIGN  /* >>= */
    | AND_ASSIGN    /* &= */
    | XOR_ASSIGN    /* ^= */
    | OR_ASSIGN     /* |= */
    ;

expression
    : assignment_expression
    | expression ',' assignment_expression      /* a = 1, b = 2 */
    ;

constant_expression
    : conditional_expression                   /* 5 + 3 */
    ;

declaration
    : declaration_specifiers ';'               /* int; */
    | declaration_specifiers init_declarator_list ';' /* int a = 5; */
    ;

declaration_specifiers
    : storage_class_specifier
    | storage_class_specifier declaration_specifiers /* static int */
    | type_specifier
    | type_specifier declaration_specifiers         /* int long */
    | type_qualifier
    | type_qualifier declaration_specifiers         /* const int */
    ;

init_declarator_list
    : init_declarator
    | init_declarator_list ',' init_declarator      /* a=1, b=2 */
    ;

init_declarator
    : declarator                 /* int a; */
    | declarator '=' initializer /* int a = 5; */
    ;

storage_class_specifier
    : TYPEDEF    /* typedef int myint; */
    | EXTERN     /* extern int x; */
    | STATIC     /* static int x; */
    | AUTO       /* auto int x; */
    | REGISTER   /* register int x; */
    ;

type_specifier
    : VOID      /* void */
    | CHAR      /* char */
    | SHORT     /* short */
    | INT       /* int */
    | LONG      /* long */
    | FLOAT     /* float */
    | DOUBLE    /* double */
    | SIGNED    /* signed */
    | UNSIGNED  /* unsigned */
    | struct_or_union_specifier /* struct S { int a; } */
    | enum_specifier            /* enum E { A, B } */
    | TYPE_NAME                 /* typedef_name */
    ;

struct_or_union_specifier
    : struct_or_union IDENTIFIER '{' struct_declaration_list '}' /* struct S { int a; } */
    | struct_or_union '{' struct_declaration_list '}'            /* struct { int a; } */
    | struct_or_union IDENTIFIER                                /* struct S */
    ;

struct_or_union
    : STRUCT   /* struct */
    | UNION    /* union */
    ;

struct_declaration_list
    : struct_declaration
    | struct_declaration_list struct_declaration
    ;

struct_declaration
    : specifier_qualifier_list struct_declarator_list ';' /* int a; */
    ;

specifier_qualifier_list
    : type_specifier specifier_qualifier_list
    | type_specifier
    | type_qualifier specifier_qualifier_list
    | type_qualifier
    ;

struct_declarator_list
    : struct_declarator
    | struct_declarator_list ',' struct_declarator /* a, b */
    ;

struct_declarator
    : declarator                  /* int a */
    | ':' constant_expression    /* :3 */
    | declarator ':' constant_expression /* a:3 */
    ;

enum_specifier
    : ENUM '{' enumerator_list '}'              /* enum {A, B} */
    | ENUM IDENTIFIER '{' enumerator_list '}'   /* enum E {A, B} */
    | ENUM IDENTIFIER                           /* enum E */
    ;

enumerator_list
    : enumerator
    | enumerator_list ',' enumerator            /* A, B */
    ;

enumerator
    : IDENTIFIER                    /* A */
    | IDENTIFIER '=' constant_expression /* A = 1 */
    ;

type_qualifier
    : CONST     /* const */
    | VOLATILE  /* volatile */
    ;

declarator
    : pointer direct_declarator     /* *p */
    | direct_declarator             /* a */
    ;

direct_declarator
    : IDENTIFIER                    /* a */
    | '(' declarator ')'            /* ( *p ) */
    | direct_declarator '[' constant_expression ']' /* a[10] */
    | direct_declarator '[' ']'     /* a[] */
    | direct_declarator '(' parameter_type_list ')' /* f(int a) */
    | direct_declarator '(' identifier_list ')'     /* f(a,b) */
    | direct_declarator '(' ')'     /* f() */
    ;

pointer
    : '*'                           /* * */
    | '*' type_qualifier_list       /* *const */
    | '*' pointer                   /* ** */
    | '*' type_qualifier_list pointer /* *const * */
    ;

type_qualifier_list
    : type_qualifier
    | type_qualifier_list type_qualifier /* const volatile */
    ;

parameter_type_list
    : parameter_list
    | parameter_list ',' ELLIPSIS   /* int a, ... */
    ;

parameter_list
    : parameter_declaration
    | parameter_list ',' parameter_declaration /* int a, float b */
    ;

parameter_declaration
    : declaration_specifiers declarator         /* int a */
    | declaration_specifiers abstract_declarator /* int * */
    | declaration_specifiers                    /* int */
    ;

identifier_list
    : IDENTIFIER
    | identifier_list ',' IDENTIFIER            /* a, b */
    ;

type_name
    : specifier_qualifier_list
    | specifier_qualifier_list abstract_declarator /* int * */
    ;

abstract_declarator
    : pointer
    | direct_abstract_declarator
    | pointer direct_abstract_declarator
    ;

direct_abstract_declarator
    : '(' abstract_declarator ')'
    | '[' ']'                      /* [] */
    | '[' constant_expression ']' /* [10] */
    | direct_abstract_declarator '[' ']' 
    | direct_abstract_declarator '[' constant_expression ']'
    | '(' ')'                      /* () */
    | '(' parameter_type_list ')'  /* (int a) */
    | direct_abstract_declarator '(' ')'
    | direct_abstract_declarator '(' parameter_type_list ')'
    ;

initializer
    : assignment_expression        /* = 5 */
    | '{' initializer_list '}'     /* {1,2} */
    | '{' initializer_list ',' '}' /* {1,2,} */
    ;

initializer_list
    : initializer
    | initializer_list ',' initializer /* 1,2,3 */
    ;

statement
    : labeled_statement
    | compound_statement
    | expression_statement
    | selection_statement
    | iteration_statement
    | jump_statement
    ;

labeled_statement
    : IDENTIFIER ':' statement         /* label: x=1; */
    | CASE constant_expression ':' statement /* case 1: */
    | DEFAULT ':' statement            /* default: */
    ;

compound_statement
    : '{' '}'                          /* {} */
    | '{' statement_list '}'           /* { x=1; } */
    | '{' declaration_list '}'         /* { int a; } */
    | '{' declaration_list statement_list '}' /* { int a; a=1; } */
    ;

declaration_list
    : declaration
    | declaration_list declaration
    ;

statement_list
    : statement
    | statement_list statement
    ;

expression_statement
    : ';'                  /* ; */
    | expression ';'       /* x=1; */
    ;

selection_statement
    : IF '(' expression ')' statement              /* if(x) y=1; */
    | IF '(' expression ')' statement ELSE statement /* if(x) y=1; else y=2; */
    | SWITCH '(' expression ')' statement          /* switch(x){...} */
    ;

iteration_statement
    : WHILE '(' expression ')' statement           /* while(x) y++; */
    | DO statement WHILE '(' expression ')' ';'    /* do y++; while(x); */
    | FOR '(' expression_statement expression_statement ')' statement /* for(;;) */
    | FOR '(' expression_statement expression_statement expression ')' statement /* for(i=0;i<10;i++) */
    ;

jump_statement
    : GOTO IDENTIFIER ';'  /* goto label; */
    | CONTINUE ';'         /* continue; */
    | BREAK ';'            /* break; */
    | RETURN ';'           /* return; */
    | RETURN expression ';'/* return x; */
    ;

translation_unit
    : external_declaration
    | translation_unit external_declaration
    ;

external_declaration
    : function_definition
    | declaration
    ;

function_definition
    : declaration_specifiers declarator declaration_list compound_statement /* int f() { int a; } */
    | declaration_specifiers declarator compound_statement                  /* int f() {} */
    | declarator declaration_list compound_statement                        /* f() { int a; } */
    | declarator compound_statement                                         /* f() {} */
    ;
